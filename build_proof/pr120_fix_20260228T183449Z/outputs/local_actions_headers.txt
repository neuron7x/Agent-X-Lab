=== .github/actions/pin-pip/action.yml
name: Pin pip
description: Deterministically pin pip version (CI-safe, evidence-friendly)
inputs:
  pip-version:
    description: Exact pip version
    required: true
runs:
  using: composite
  steps:
    - name: Pin pip version
      shell: bash
      run: |
        set -euo pipefail
        python -m pip install --upgrade "pip==${{ inputs.pip-version }}"
        python -m pip --version
=== .github/actions/setup-python-pip/action.yml
name: Setup Python + pinned pip
description: Install Python and pin pip to a required version.
inputs:
  python-version:
    required: true
  pip-version:
    required: true
runs:
  using: composite
  steps:
    - name: Setup Python
      uses: actions/setup-python@82c7e631bb3cdc910f68e0081d67478d79c6982d # v5.1.0
      with:
        python-version: ${{ inputs.python-version }}
    - name: Pin pip
      uses: ./.github/actions/pin-pip
      with:
        pip-version: ${{ inputs.pip-version }}
